# 9. 地图与寻路系统

## 9.1 地图管理器

### 9.1.1 地图数据结构

地图系统采用网格化设计，每个格子（Block）代表地图上的一个位置，支持不同类型的地图元素和动态状态更新。

```csharp
// 地图块类型枚举
public enum BlockType
{
    Null = 0,      // 空地
    Obstacle = 1,  // 障碍物
    Start = 2,     // 起始点
    End = 3,       // 终点
    Trap = 4,      // 陷阱
    Treasure = 5,  // 宝藏
}

// 地图块类
public class Block
{
    public int rowIndex;           // 行索引
    public int colIndex;           // 列索引
    public BlockType type;         // 块类型
    public GameObject gameObject;  // 关联的游戏对象
    public bool isPassable;        // 是否可通过
    public int movementCost;       // 移动消耗
    public List<Block> neighbors; // 相邻块

    public Block(int row, int col, BlockType blockType = BlockType.Null)
    {
        rowIndex = row;
        colIndex = col;
        type = blockType;
        neighbors = new List<Block>();
        UpdatePassability();
    }

    private void UpdatePassability()
    {
        isPassable = type != BlockType.Obstacle;
        movementCost = type == BlockType.Trap ? 3 : 1;
    }

    public void SetType(BlockType newType)
    {
        type = newType;
        UpdatePassability();

        // 触发地图更新事件
        GameApp.EventCenter.BroadcastEvent("BlockTypeChanged", this);
    }

    public Vector3 GetWorldPosition()
    {
        return new Vector3(colIndex, rowIndex, 0);
    }

    public float GetDistance(Block other)
    {
        int dx = Mathf.Abs(rowIndex - other.rowIndex);
        int dy = Mathf.Abs(colIndex - other.colIndex);
        return Mathf.Sqrt(dx * dx + dy * dy);
    }
}
```

### 9.1.2 地图管理器实现

```csharp
public class MapManager
{
    private Block[,] _grid;                    // 地图网格
    private int _width;                        // 地图宽度
    private int _height;                       // 地图高度
    private Dictionary<BlockType, Color> _blockColors; // 块类型颜色映射

    public int Width => _width;
    public int Height => _height;

    public MapManager()
    {
        InitializeBlockColors();
    }

    // 初始化地图
    public void InitializeMap(int width, int height)
    {
        _width = width;
        _height = height;
        _grid = new Block[height, width];

        // 创建地图网格
        for (int row = 0; row < height; row++)
        {
            for (int col = 0; col < width; col++)
            {
                _grid[row, col] = new Block(row, col);
            }
        }

        // 建立邻居关系
        BuildNeighborRelationships();

        // 生成地图视觉效果
        GenerateMapVisuals();
    }

    // 获取指定位置的地图块
    public Block GetBlock(int row, int col)
    {
        if (row >= 0 && row < _height && col >= 0 && col < _width)
        {
            return _grid[row, col];
        }
        return null;
    }

    // 改变地图块类型
    public void ChangeBlockType(int row, int col, BlockType newType)
    {
        Block block = GetBlock(row, col);
        if (block != null)
        {
            BlockType oldType = block.type;
            block.SetType(newType);

            // 更新视觉效果
            UpdateBlockVisual(block, oldType, newType);

            // 触发地图更新事件
            GameApp.EventCenter.BroadcastEvent("MapBlockChanged", new object[] { block, oldType, newType });
        }
    }

    // 检查位置是否可通过
    public bool IsPositionPassable(int row, int col)
    {
        Block block = GetBlock(row, col);
        return block != null && block.isPassable;
    }

    // 获取可移动到的邻居位置
    public List<Block> GetPassableNeighbors(Block block)
    {
        return block.neighbors.Where(n => n.isPassable).ToList();
    }

    // 生成地图视觉效果
    private void GenerateMapVisuals()
    {
        for (int row = 0; row < _height; row++)
        {
            for (int col = 0; col < _width; col++)
            {
                Block block = _grid[row, col];
                CreateBlockVisual(block);
            }
        }
    }

    private void CreateBlockVisual(Block block)
    {
        // 创建地图块的可视化对象
        GameObject blockObj = GameObject.CreatePrimitive(PrimitiveType.Cube);
        blockObj.transform.position = block.GetWorldPosition();
        blockObj.transform.localScale = new Vector3(0.9f, 0.9f, 0.1f);

        // 设置颜色
        Renderer renderer = blockObj.GetComponent<Renderer>();
        renderer.material.color = _blockColors[block.type];

        // 关联游戏对象
        block.gameObject = blockObj;
    }

    private void UpdateBlockVisual(Block block, BlockType oldType, BlockType newType)
    {
        if (block.gameObject != null)
        {
            Renderer renderer = block.gameObject.GetComponent<Renderer>();
            renderer.material.color = _blockColors[newType];
        }
    }

    private void BuildNeighborRelationships()
    {
        for (int row = 0; row < _height; row++)
        {
            for (int col = 0; col < _width; col++)
            {
                Block current = _grid[row, col];
                current.neighbors.Clear();

                // 添加四个方向的邻居（可以扩展为八个方向）
                AddNeighborIfValid(current, row - 1, col); // 上
                AddNeighborIfValid(current, row + 1, col); // 下
                AddNeighborIfValid(current, row, col - 1); // 左
                AddNeighborIfValid(current, row, col + 1); // 右
            }
        }
    }

    private void AddNeighborIfValid(Block current, int row, int col)
    {
        Block neighbor = GetBlock(row, col);
        if (neighbor != null)
        {
            current.neighbors.Add(neighbor);
        }
    }

    private void InitializeBlockColors()
    {
        _blockColors = new Dictionary<BlockType, Color>
        {
            { BlockType.Null, Color.white },
            { BlockType.Obstacle, Color.black },
            { BlockType.Start, Color.green },
            { BlockType.End, Color.red },
            { BlockType.Trap, Color.yellow },
            { BlockType.Treasure, Color.blue }
        };
    }

    // 清空地图
    public void Clear()
    {
        if (_grid != null)
        {
            for (int row = 0; row < _height; row++)
            {
                for (int col = 0; col < _width; col++)
                {
                    if (_grid[row, col].gameObject != null)
                    {
                        GameObject.Destroy(_grid[row, col].gameObject);
                    }
                }
            }
        }
        _grid = null;
    }
}
```

## 9.2 A*寻路算法

### 9.2.1 A*算法实现

```csharp
public class AStar
{
    private class Node
    {
        public Block block;           // 对应的地图块
        public Node parent;          // 父节点
        public float gCost;          // 从起点到当前节点的实际代价
        public float hCost;          // 从当前节点到终点的估计代价
        public float fCost => gCost + hCost; // 总代价

        public Node(Block block, Node parent = null)
        {
            this.block = block;
            this.parent = parent;
            this.gCost = 0;
            this.hCost = 0;
        }
    }

    public static List<Block> FindPath(Block start, Block end)
    {
        if (start == null || end == null || start == end)
            return null;

        if (!start.isPassable || !end.isPassable)
            return null;

        var openSet = new List<Node>();     // 开放列表
        var closedSet = new HashSet<Block>(); // 关闭列表
        var nodeGrid = new Dictionary<Block, Node>(); // 节点映射

        Node startNode = new Node(start);
        openSet.Add(startNode);
        nodeGrid[start] = startNode;

        while (openSet.Count > 0)
        {
            // 找到fCost最小的节点
            Node currentNode = GetLowestFCostNode(openSet);

            // 如果到达终点
            if (currentNode.block == end)
            {
                return RetracePath(startNode, currentNode);
            }

            // 从开放列表移除，添加到关闭列表
            openSet.Remove(currentNode);
            closedSet.Add(currentNode.block);

            // 检查所有邻居
            foreach (Block neighbor in currentNode.block.neighbors)
            {
                if (!neighbor.isPassable || closedSet.Contains(neighbor))
                    continue;

                float newMovementCost = currentNode.gCost + GetDistance(currentNode.block, neighbor) * neighbor.movementCost;

                Node neighborNode;
                if (!nodeGrid.ContainsKey(neighbor))
                {
                    neighborNode = new Node(neighbor);
                    nodeGrid[neighbor] = neighborNode;
                }
                else
                {
                    neighborNode = nodeGrid[neighbor];
                }

                if (newMovementCost < neighborNode.gCost || !openSet.Contains(neighborNode))
                {
                    neighborNode.gCost = newMovementCost;
                    neighborNode.hCost = GetDistance(neighbor, end);
                    neighborNode.parent = currentNode;

                    if (!openSet.Contains(neighborNode))
                    {
                        openSet.Add(neighborNode);
                    }
                }
            }
        }

        // 未找到路径
        return null;
    }

    private static Node GetLowestFCostNode(List<Node> openSet)
    {
        Node lowestNode = openSet[0];
        for (int i = 1; i < openSet.Count; i++)
        {
            if (openSet[i].fCost < lowestNode.fCost)
            {
                lowestNode = openSet[i];
            }
        }
        return lowestNode;
    }

    private static List<Block> RetracePath(Node startNode, Node endNode)
    {
        var path = new List<Block>();
        Node currentNode = endNode;

        while (currentNode != startNode)
        {
            path.Add(currentNode.block);
            currentNode = currentNode.parent;
        }

        path.Reverse();
        return path;
    }

    private static float GetDistance(Block a, Block b)
    {
        int dstX = Mathf.Abs(a.colIndex - b.colIndex);
        int dstY = Mathf.Abs(a.rowIndex - b.rowIndex);

        if (dstX > dstY)
            return 14f * dstY + 10f * (dstX - dstY);
        else
            return 14f * dstX + 10f * (dstY - dstX);
    }
}
```

### 9.2.2 寻路优化

```csharp
public class OptimizedAStar
{
    private static Dictionary<string, List<Block>> _pathCache;
    private static int _maxCacheSize = 100;

    static OptimizedAStar()
    {
        _pathCache = new Dictionary<string, List<Block>>();
    }

    public static List<Block> FindPathOptimized(Block start, Block end)
    {
        // 检查缓存
        string cacheKey = $"{start.rowIndex},{start.colIndex}_{end.rowIndex},{end.colIndex}";
        if (_pathCache.ContainsKey(cacheKey))
        {
            return new List<Block>(_pathCache[cacheKey]);
        }

        // 计算路径
        List<Block> path = AStar.FindPath(start, end);

        // 缓存结果
        if (path != null)
        {
            CachePath(cacheKey, path);
        }

        return path;
    }

    private static void CachePath(string key, List<Block> path)
    {
        // 清理缓存（如果超出限制）
        if (_pathCache.Count >= _maxCacheSize)
        {
            var firstKey = _pathCache.Keys.First();
            _pathCache.Remove(firstKey);
        }

        _pathCache[key] = new List<Block>(path);
    }

    public static void ClearCache()
    {
        _pathCache.Clear();
    }
}
```

## 9.3 BFS算法实现

### 9.3.1 广度优先搜索

```csharp
public class BFS
{
    public static List<Block> FindPath(Block start, Block end)
    {
        if (start == null || end == null || start == end)
            return null;

        if (!start.isPassable || !end.isPassable)
            return null;

        var queue = new Queue<Block>();
        var visited = new HashSet<Block>();
        var parent = new Dictionary<Block, Block>();

        queue.Enqueue(start);
        visited.Add(start);

        while (queue.Count > 0)
        {
            Block current = queue.Dequeue();

            if (current == end)
            {
                return ReconstructPath(parent, start, end);
            }

            foreach (Block neighbor in current.neighbors)
            {
                if (neighbor.isPassable && !visited.Contains(neighbor))
                {
                    queue.Enqueue(neighbor);
                    visited.Add(neighbor);
                    parent[neighbor] = current;
                }
            }
        }

        return null; // 未找到路径
    }

    private static List<Block> ReconstructPath(Dictionary<Block, Block> parent, Block start, Block end)
    {
        var path = new List<Block>();
        Block current = end;

        while (current != start)
        {
            path.Add(current);
            current = parent[current];
        }

        path.Add(start);
        path.Reverse();
        return path;
    }

    // 查找指定范围内的所有可达位置
    public static List<Block> FindReachableBlocks(Block start, int maxDistance)
    {
        var reachable = new List<Block>();
        var queue = new Queue<(Block block, int distance)>();
        var visited = new HashSet<Block>();

        queue.Enqueue((start, 0));
        visited.Add(start);

        while (queue.Count > 0)
        {
            var (current, distance) = queue.Dequeue();

            if (distance <= maxDistance)
            {
                reachable.Add(current);
            }

            if (distance < maxDistance)
            {
                foreach (Block neighbor in current.neighbors)
                {
                    if (neighbor.isPassable && !visited.Contains(neighbor))
                    {
                        queue.Enqueue((neighbor, distance + 1));
                        visited.Add(neighbor);
                    }
                }
            }
        }

        return reachable;
    }
}
```

### 9.4.1 战斗地图生成

```csharp
public class BattleMapGenerator
{
    public static void GenerateRandomMap(int width, int height, float obstacleDensity = 0.3f)
    {
        GameApp.MapManager.InitializeMap(width, height);

        // 随机生成障碍物
        for (int row = 0; row < height; row++)
        {
            for (int col = 0; col < width; col++)
            {
                if (Random.value < obstacleDensity)
                {
                    GameApp.MapManager.ChangeBlockType(row, col, BlockType.Obstacle);
                }
            }
        }

        // 确保有可通行的路径
        EnsurePathExists();

        // 放置敌人
        PlaceEnemies();
    }

    private static void EnsurePathExists()
    {
        // 从左上角到右下角确保有路径
        Block start = GameApp.MapManager.GetBlock(0, 0);
        Block end = GameApp.MapManager.GetBlock(
            GameApp.MapManager.Height - 1,
            GameApp.MapManager.Width - 1
        );

        List<Block> path = AStar.FindPath(start, end);
        if (path == null)
        {
            // 如果没有路径，清除一些障碍物
            ClearObstaclesForPath(start, end);
        }
    }

    private static void PlaceEnemies()
    {
        int enemyCount = Random.Range(3, 8);

        for (int i = 0; i < enemyCount; i++)
        {
            Block enemyBlock = FindRandomEmptyBlock();
            if (enemyBlock != null)
            {
                SpawnEnemy(enemyBlock);
            }
        }
    }

    private static Block FindRandomEmptyBlock()
    {
        List<Block> emptyBlocks = new List<Block>();

        for (int row = 0; row < GameApp.MapManager.Height; row++)
        {
            for (int col = 0; col < GameApp.MapManager.Width; col++)
            {
                Block block = GameApp.MapManager.GetBlock(row, col);
                if (block.type == BlockType.Null)
                {
                    emptyBlocks.Add(block);
                }
            }
        }

        return emptyBlocks.Count > 0 ? emptyBlocks[Random.Range(0, emptyBlocks.Count)] : null;
    }

    private static void SpawnEnemy(Block block)
    {
        // 创建敌人预制体
        GameObject enemyObj = GameObject.Instantiate(
            Resources.Load<GameObject>("Model/orc"),
            block.GetWorldPosition(),
            Quaternion.identity
        );

        Enemy enemy = enemyObj.AddComponent<Enemy>();
        enemy.Init(new Dictionary<string, string> { {"Type", "Orc"} }, block);

        // 设置地图块为障碍物
        GameApp.MapManager.ChangeBlockType(block.rowIndex, block.colIndex, BlockType.Obstacle);
    }
}
```

### 9.4.2 移动范围计算

```csharp
public class MovementCalculator
{
    // 计算角色可以移动到的所有位置
    public static List<Block> CalculateMovementRange(ModelBase character, float moveRange)
    {
        var reachableBlocks = new List<Block>();
        var queue = new Queue<(Block block, float remainingMove)>();
        var visited = new HashSet<Block>();

        Block startBlock = GameApp.MapManager.GetBlock(character.rowIndex, character.colIndex);
        queue.Enqueue((startBlock, moveRange));
        visited.Add(startBlock);

        while (queue.Count > 0)
        {
            var (current, remaining) = queue.Dequeue();

            if (remaining >= 0 && current != startBlock)
            {
                reachableBlocks.Add(current);
            }

            if (remaining > 0)
            {
                foreach (Block neighbor in current.neighbors)
                {
                    if (neighbor.isPassable && !visited.Contains(neighbor))
                    {
                        float moveCost = neighbor.movementCost;
                        if (remaining >= moveCost)
                        {
                            queue.Enqueue((neighbor, remaining - moveCost));
                            visited.Add(neighbor);
                        }
                    }
                }
            }
        }

        return reachableBlocks;
    }

    // 显示移动范围
    public static void ShowMovementRange(List<Block> reachableBlocks)
    {
        foreach (Block block in reachableBlocks)
        {
            if (block.gameObject != null)
            {
                // 高亮显示可移动位置
                Renderer renderer = block.gameObject.GetComponent<Renderer>();
                renderer.material.color = Color.cyan;
            }
        }
    }

    // 隐藏移动范围
    public static void HideMovementRange(List<Block> reachableBlocks)
    {
        foreach (Block block in reachableBlocks)
        {
            if (block.gameObject != null)
            {
                // 恢复原始颜色
                Renderer renderer = block.gameObject.GetComponent<Renderer>();
                renderer.material.color = GetBlockTypeColor(block.type);
            }
        }
    }

    private static Color GetBlockTypeColor(BlockType type)
    {
        return type switch
        {
            BlockType.Null => Color.white,
            BlockType.Obstacle => Color.black,
            BlockType.Start => Color.green,
            BlockType.End => Color.red,
            BlockType.Trap => Color.yellow,
            BlockType.Treasure => Color.blue,
            _ => Color.white
        };
    }
}
```

## 9.5 性能优化

### 9.5.1 地图更新优化

```csharp
public class MapUpdateOptimizer
{
    private List<Block> _dirtyBlocks; // 需要更新的块
    private float _updateInterval;    // 更新间隔
    private float _timer;            // 计时器

    public MapUpdateOptimizer()
    {
        _dirtyBlocks = new List<Block>();
        _updateInterval = 0.1f; // 100ms更新一次
        _timer = 0f;
    }

    public void Update(float dt)
    {
        _timer += dt;

        if (_timer >= _updateInterval && _dirtyBlocks.Count > 0)
        {
            BatchUpdateBlocks();
            _timer = 0f;
        }
    }

    public void MarkBlockDirty(Block block)
    {
        if (!_dirtyBlocks.Contains(block))
        {
            _dirtyBlocks.Add(block);
        }
    }

    private void BatchUpdateBlocks()
    {
        // 批量更新所有脏块
        foreach (Block block in _dirtyBlocks)
        {
            UpdateBlockVisual(block);
        }

        _dirtyBlocks.Clear();
    }

    private void UpdateBlockVisual(Block block)
    {
        // 更新块的视觉效果
        if (block.gameObject != null)
        {
            Renderer renderer = block.gameObject.GetComponent<Renderer>();
            renderer.material.color = GetBlockTypeColor(block.type);
        }
    }
}
```
